Step 3: Load the DHS Data

Household Recode (HR file)

# Select relevant household variables
vars_hr <- c("hhid", "hv001", "hv002", "hv005",  # cluster, household ID, weight
             "hv008", "hv009", "hv024")          # date of interview, household size, region

# Load the household recode file from the Raw folder
hr_raw <- read_dta("Raw/MZHR81FL.DTA",
                   col_select = all_of(vars_hr))

Individual Recode (IR file)

# Select relevant individual variables
vars_ir <- c("v001", "v002", "v005",              # cluster, household ID, individual weight
             "v008", "v011",                      # date of interview, date of birth
             "v012", "v013", "v028", "v149")      # age, 5-year age group, sample line, education

# Load the individual recode file
ir_raw <- read_dta("Raw/MZIR81FL.DTA",
                   col_select = all_of(vars_ir))

Step 4: Merge the Data

dhs_merged <- ir_raw %>% # Inner_join to keep only observations with matches
  inner_join(hr_raw, by = c("v001" = "hv001", "v002" = "hv002")) # Different variable names in each dataset

Step 6: Recode Education


# Check the values of v149 (highest education level)
dhs_merged$v149 %>% unique()

# Create binary education group
dhs_merged %>% 
  transmute(v149, 
            ed_group = case_when(
              v149 < 3 ~ "No education or primary",
              v149 >= 3 ~ "Some secondary or higher"
            ))  %>% table()

Step 7: Make a Histogram

dhs_merged %>%
  mutate(
    age_years = (v008 - v011) / 12,  # Calculate age
    ed_group = case_when(
      v149 < 3 ~ "No or Primary",
      v149 >= 3 ~ "Secondary or Higher"
    )) %>%
  ggplot(aes(x = age_years, fill = ed_group)) +
  geom_histogram(binwidth = 1, position = "identity", alpha = 0.5) +
  labs(title = "Age Distribution by Education Group")

Step 8: Weighted Summary by Region


dhs_merged %>%
  mutate(
    region = as_factor(hv024),                 # Convert region to factor
    age_years = (v008 - v011) / 12,
    ed_group = case_when(                      # Recode education
      v149 < 3 ~ "No or Primary",
      v149 >= 3 ~ "Secondary or Higher"
    ),
    weight = v005 / 1e6                        # Apply DHS sample weight (divide by 1,000,000)
  ) %>%
  group_by(region, ed_group) %>%
  summarise(wt_n = sum(weight, na.rm = TRUE), .groups = "drop") %>%
  group_by(region) %>%
  mutate(prop = wt_n / sum(wt_n)) %>%
  ggplot(aes(x = region, y = prop, fill = ed_group)) +
  geom_col(position = "fill") +  # Stacked bar chart scaled to 100%
  scale_y_continuous(labels = scales::percent) +
  labs(
    x = "Region", y = "Proportion", fill = "Education Group",
    title = "Education Level by Region (Weighted by Individual)"
  ) +
  theme_minimal()
